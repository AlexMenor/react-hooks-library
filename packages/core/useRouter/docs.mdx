# useRouter

Reactive browser location.

## Usage

```tsx
import { useRouter } from '@react-hooks-library/core'

export function Demo() {
  const router = useRouter()

  return (
    <div>
      <pre>
        <code>{JSON.stringify(router, null, 2)}</code>
      </pre>
    </div>
  )
}
```
