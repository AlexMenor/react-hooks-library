# useNetwork

Reactive Network status.

## Usage

```tsx
import { useNetwork } from '@react-hooks-library/core'

export function Demo() {
  const network = useNetwork()

  return (
    <div>
      <dl className="my-4">
        {Object.entries(network).map(([key, value]) => (
          <div key={`${key} + ${value}`}>
            <dt>{key}:</dt>
            <dd>{`${value}` ?? 'unknown'}</dd>
          </div>
        ))}
      </dl>
    </div>
  )
}
```
