# useMount

Run a function when a component is mounted.

## Usage

```tsx
import { useMount } from '@react-hooks-library/core'

function Demo() {
  useMount(() => {
    alert('I have been mounted')
  })

  return <div></div>
}

```

## Type Declarations

```typescript
/**
 * Run a function when a component is mounted.
 *
 * @param callback function to be executed
 */
```
