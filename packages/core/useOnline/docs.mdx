# useOnline

Reactive online status.

## Usage

```tsx
import { useOnline } from '@react-hooks-library/core'

export function Demo() {
  const isOnline = useOnline()

  return (
    <div>
      {isOnline ? <div>Online</div> : <div>Offline</div>}
      <small>Toggle your network to observe changes</small>
    </div>
  )
}
```
